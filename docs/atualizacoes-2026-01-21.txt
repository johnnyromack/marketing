--- Migration 2: Distribuição Mensal de Orçamento de Área ---
Arquivo: supabase/migrations/20260121002524_0e30b0d4-4910-414b-899e-4dd9b4379f4f.sql

```sql
CREATE TABLE public.orcamento_area_distribuicao (
  id UUID NOT NULL DEFAULT gen_random_uuid() PRIMARY KEY,
  orcamento_id UUID NOT NULL REFERENCES public.orcamentos(id) ON DELETE CASCADE,
  ano INTEGER NOT NULL,
  mes INTEGER NOT NULL CHECK (mes >= 1 AND mes <= 12),
  valor_orcado NUMERIC NOT NULL DEFAULT 0,
  verba_extra NUMERIC NOT NULL DEFAULT 0,
  observacoes TEXT,
  created_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT now(),
  updated_at TIMESTAMP WITH TIME ZONE NOT NULL DEFAULT now(),
  UNIQUE(orcamento_id, ano, mes)
);

ALTER TABLE public.orcamento_area_distribuicao ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Authenticated users can view orcamento_area_distribuicao"
ON public.orcamento_area_distribuicao FOR SELECT TO authenticated USING (true);

CREATE POLICY "Authenticated users can insert orcamento_area_distribuicao"
ON public.orcamento_area_distribuicao FOR INSERT TO authenticated WITH CHECK (true);

CREATE POLICY "Authenticated users can update orcamento_area_distribuicao"
ON public.orcamento_area_distribuicao FOR UPDATE TO authenticated USING (true);

CREATE POLICY "Authenticated users can delete orcamento_area_distribuicao"
ON public.orcamento_area_distribuicao FOR DELETE TO authenticated USING (true);

CREATE TRIGGER update_orcamento_area_distribuicao_updated_at
BEFORE UPDATE ON public.orcamento_area_distribuicao
FOR EACH ROW
EXECUTE FUNCTION public.update_updated_at_column();
```

  * Observações por mês
- Validações:
  * Total distribuído vs orçamento aprovado
  * Indicador visual de status

ESTRUTURA DE DADOS

tipos_custo:
- id: UUID (PK)
- nome: TEXT (unique)
- descricao: TEXT (nullable)
- tipo_orcamento: TEXT ('proprio' | 'compartilhado')
- ativo: BOOLEAN (default true)
- created_at: TIMESTAMPTZ
- updated_at: TIMESTAMPTZ

orcamento_area_distribuicao:
- id: UUID (PK)
- orcamento_id: UUID (FK -> orcamentos)
- ano: INTEGER
- mes: INTEGER (1-12)
- valor_orcado: NUMERIC (default 0)
- verba_extra: NUMERIC (default 0)
- observacoes: TEXT (nullable)
- created_at: TIMESTAMPTZ
- updated_at: TIMESTAMPTZ
- UNIQUE(orcamento_id, ano, mes)
